1강: 자바스크립트란?

자바스크립트는 웹페이지를 동적으로 만들어주는 프로그래밍 언어입니다.

// 첫 번째 자바스크립트 코드 console.log("안녕하세요, JavaScript!"); alert("환영합니다!");

2강: 변수 선언하기

변수는 데이터를 저장하는 공간입니다.

let name = "홍길동"; const age = 25; var city = "서울"; console.log(name); console.log(age); console.log(city);

3강: 데이터 타입

자바스크립트의 다양한 데이터 타입을 알아봅시다.

// 문자열 (String) let message = "안녕하세요"; // 숫자 (Number) let score = 100; // 불린 (Boolean) let isStudent = true; // 배열 (Array) let colors = ["빨강", "파랑", "초록"];

4강: 함수 만들기

함수는 특정 작업을 수행하는 코드 블록입니다.

function greet(name) { return "안녕하세요, " + name + "님!"; } function add(a, b) { return a + b; } console.log(greet("김철수")); console.log(add(10, 20));

5강: 조건문 (if-else)

조건에 따라 다른 코드를 실행합니다.

let score = 85; if (score >= 90) { console.log("A 등급"); } else if (score >= 80) { console.log("B 등급"); } else { console.log("더 열심히 공부하세요!"); }

6강: 반복문 (for, while)

같은 작업을 여러 번 반복할 때 사용합니다.

// for 반복문 for (let i = 1; i <= 5; i++) { console.log(i + "번째 반복"); } // 배열 반복 let fruits = ["사과", "바나나", "오렌지"]; for (let fruit of fruits) { console.log(fruit); }
