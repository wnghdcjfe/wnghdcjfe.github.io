function solution(food_times, k) {
    let list = [];
    let max = 0;
    const length = food_times.length;
    for(let i = 0; i < food_times.length; i++) {
        list.push({ value: food_times[i], index: i + 1 });
    }
    list.sort((a, b) => a.value - b.value);

    for(let i = 0; i < list.length; i++) {
        const pass = (list.length - i) * (list[i].value - (list[i - 1] ? list[i - 1].value : 0));
        if(k < pass) {
            break;
        } else {
            max = list[i].value;
            k = k - pass;
        }
    }

    list = list.filter(v => (v.value - max) > 0);
    list.sort((a, b) => a.index - b.index);

    if(list.length === 0) {
        return -1;
    } else if (list.reduce((acc, v) => acc + v.value, 0) < k){
        return -1;
    }

    return list[k % list.length].index;
}

#include <string>
#include <vector>
#include <algorithm>

using namespace std;

bool compare(pair<int, int>& a, pair<int, int>& b) {
	return a.second < b.second;
}

int solution(vector<int> food_times, long long k) {

	int answer = -1;
	int size = food_times.size();

	vector<pair<int, int>> timesAndIndex(size);
	for (int i = 0; i < size; i++) {
		timesAndIndex[i] = { food_times[i], i };
	}
	sort(timesAndIndex.begin(), timesAndIndex.end());

	int row = 0;
	int before = 0;
	while (row < size) {
		int time = timesAndIndex[row].first;
		int startInd = row;
		while (row + 1 < size && timesAndIndex[row + 1].first == time) {
			row++;
		}
		int cnt = size - startInd;

		if (k < (long long int)(time - before) * (long long int)cnt) {
			sort(timesAndIndex.begin() + startInd, timesAndIndex.end(), compare);
			answer = timesAndIndex[startInd + (k % cnt)].second + 1; 
			break;
		}

		k -= (time - before) * cnt;
		before = time;
		row++;
	}

	return answer;
}
